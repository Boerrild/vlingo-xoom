@Queries(protocol = Queries.class, actor = QueriesActor.class)
@Model(protocol = Greeting.class, actor = GreetingEntity.class, data = GreetingData.class)
public interface GreetingResource {

    @Route(method = POST, path="/greetings", handler = "defineWith(stage, data.message, data.description)")
    @Response(data = "GreetingData.from")
    Completes<Response> defineGreeting(@Body GreetingData data);

    @Route(method = PATCH, path = "/greetings/{greetingId}/message", handler = "withMessage(message.value)")
    @Response(data = "GreetingData.from")  
    Completes<Response> changeGreetingMessage(@Id String greetingId, @Body MessageData message);

    @Route(method = PATCH, path = "/greetings/{greetingId}/description", handler = "withDescription(description.value)")
    @Response(data = "GreetingData.from")  
    Completes<Response> changeGreetingDescription(@Id String greetingId, @Body DescriptionData description);

    @Route(method = GET, path = "/greetings", handler="greeting")
    Completes<Response> queryGreetings();

    @Route(method = GET, path = "/greetings/{greetingId}", handler="greetingOf(greetingId)")
    Completes<Response> queryGreeting(String greetingId);

}
